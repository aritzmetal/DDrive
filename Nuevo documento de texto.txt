<?xml version="1.0" encoding="UTF-8"?>
 <!-- Set the project name, basedir and default target to be executed-->
<project name="Aplicacion Cliente de DDrive" default="help" basedir=".">
  <!-- Sets the properties here-->
  <property name="rmiregistry.port" value="1050"/>
  <property name="rmiregistry.IP" value="127.0.0.1"/>
  <property name="src.dir" location="src" />
  <property name="build.dir" location="bin" />
 
  <!-- Target for deleting the existing directories-->
  <target name="clean">
    <delete dir="${build.dir}" />
  </target>
 
  <!-- Target for creating the new directories-->
  <target name="makedir">
    <mkdir dir="${build.dir}" />
  </target>
 
  <!-- Target for compiling the java code-->
  <target name="compile" depends="clean, makedir">
    <javac srcdir="${src.dir}" destdir="${build.dir}">
    </javac>	
  </target>
 
 <target name="help">
    <echo>Usage: ant [build | client]</echo>
  </target>

 
</project>
